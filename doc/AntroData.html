def calculate_rcc               
    cintura_med = ((@cintura[0] + @cintura[1]) / 2)
    cadera_med = ((@cadera[0] + @cadera[1]) / 2)

    @rcc = (cintura_med / cadera_med).round(2)
end

def clasification_imc       
    case @imc
    when 0 .. 18.5
        "Bajo peso"
    when 18.5 .. 24.9
        "Adecuado"
    when 25 .. 29.9
        "Sobrepeso"
    when 30 .. 34.9
        "Obesidad grado 1"
    when 35 .. 39.9
        "Obesidad grado 2"
    else
        "Obesidad grado 3"
    end
end

def clasification_rcc    
    if @sexo == 1
        case @rcc
        when 0.83 .. 0.88
            "Bajo"
        when 0.88 .. 0.95
            "Moderado"
        when 0.95 .. 1.01
            "Alto"
        when 1.01 .. 2
            "Muy alto"
        end
    else
        case @rcc
        when 0.72 .. 0.75
            "Bajo"
        when 0.75 .. 0.82
            "Moderado"
        when 0.82 .. 2
            "Alto"
        end
    end
end

def to_s       
    puts "------ Valores ------"
    tmp = "Peso: #{@peso}\n"
    tmp += "Altura: #{@talla}\n"
    tmp += "Edad: #{@edad}\n"
    tmp += "Sexo: #{@sexo}\n"
    tmp += "Cincurferencia de la cintura: #{@cintura}\n"
    tmp += "Cincurferencia de la cadera: #{@cadera}\n"
    tmp += "IMC: #{self.clasification_imc }\n"
    tmp += "Porcentaje de grasa: #{self.calculate_percentgrasa}\n"
    tmp += "RCC: #{self.clasification_rcc}\n"
end
